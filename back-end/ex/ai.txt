import { Request, Response } from "express";
import { GoogleGenerativeAI, Type, FunctionCall } from "@google/generative-ai";
import dotenv from "dotenv";
import customerModal from "../models/customer-modal";

dotenv.config();

// API Key එක දීලා Gemini එක initialize කරගන්නවා
const genAI = new GoogleGenerativeAI(process.env.GEMINI_API_KEY as string);

// ── DATABASE FUNCTIONS ───────────────────────────────────────────────────

const getAllOnlineMembers = async (): Promise<number> => {
  try {
    const members = await customerModal.find({ isOnline: true });
    return members.length;
  } catch (err) {
    console.error("Database Error (getAllOnlineMembers):", err);
    return 0;
  }
};

  };
};

// ඔයා මුලින්ම ලියපු නීති රීති මාලාව සහ Agent instruction එක එකතු කර හදපු ප්‍රධාන System Prompt එක

// Tools (Functions) Declarations
const TOOLS_CONFIG = [
  {
    functionDeclarations: [
      {
        name: "getSeetuStatusOfUser",
        description: "Fetches real-time Seetu status, next payment amount, due date, and payment history for a given unique User ID.",
        parameters: {
          type: Type.OBJECT,
          properties: {
            userId: {
              type: Type.STRING,
              description: "The unique registration number or ID of the user (e.g., U123, CUST-01).",
            },
          },

        },
      },
      {
        name: "getAllOnlineMembers",

        parameters: {
          type: Type.OBJECT,
          properties: {},
        },
      },
    ],

// ── CORE LOGIC FOR RUNNING THE AGENT (REUSABLE) ──────────────────────────

/**
 * දීපු Model name එකකට අනුව මුළු Gemini process එකම handle කරන පොදු function එකක්.
 * මෙයින් කේතය නැවත නැවත ලිවීම (Code duplication) වැළකේ.
 */
const executeGeminiAgent = async (modelName: string, message: string): Promise<string> => {
  const model = genAI.getGenerativeModel({
    model: modelName,
    systemInstruction: SYSTEM_INSTRUCTION,
    tools: TOOLS_CONFIG,
  });

  // STEP A: User message එක Model එකට යවනවා
  const result = await model.generateContent(message);
  const response = result.response;
  const functionCalls = response.functionCalls as FunctionCall[] | undefined;

  // Gemini එකෙන් Function Call එකක් ආවොත් මීළඟ පියවර ක්‍රියාත්මක කරනවා
  if (functionCalls && functionCalls.length > 0) {
    const call = functionCalls[0];
    let databaseResult: any = null;

    // 1. අදාළ Function එක මොකක්ද කියලා හඳුනාගෙන Database එකෙන් ඩේටා ගන්නවා
    if (call.name === "getSeetuStatusOfUser") {
      const args = call.args as { userId: string };
      databaseResult = await getSeetuStatusOfUser(args.userId);
    } else if (call.name === "getAllOnlineMembers") {
      const count = await getAllOnlineMembers();
      databaseResult = { onlineMembersCount: count };
    }

    // 2. STEP B: Database එකෙන් ආපු දත්ත නැවත Gemini එකට දී අවසන් පිළිතුර සකස් කරගන්නවා
    if (databaseResult !== null) {
      const chat = model.startChat({
        history: [
          { role: "user", parts: [{ text: message }] },
          { role: "model", parts: [{ functionCall: call }] },
        ],
      });

      const finalResult = await chat.sendMessage([
        {
          functionResponse: {
            name: call.name,
            response: { result: databaseResult },
          },
        },
      ]);

      return finalResult.response.text();
    }
  }

  // සාමාන්‍ය ප්‍රශ්නයක් නම් (Tool එකක් අවශ්‍ය නැත්නම්) කෙලින්ම text එක දෙනවා
  return response.text();
};

// ── MAIN EXPORT CONTROLLER ───────────────────────────────────────────────

/**
 * ප්‍රධාන AI Controller එක - මෙතනින් තමයි Fallback (Pro -> Flash) එක සිද්ද වෙන්නේ
 */
export const generateContent = async (req: Request, res: Response): Promise<void> => {
  try {
    const { message } = req.body;

    if (!message) {
      res.status(400).json({ error: "Message is required" });
      return;
    }

    try {
      // 1. මුලින්ම උපරිම මොළයක් තියෙන 'gemini-1.5-pro' එකෙන් Run කරන්න උත්සාහ කරනවා
      console.log("⚡ Trying with Gemini 1.5 Pro...");
      const reply = await executeGeminiAgent("gemini-1.5-pro", message);
      
      res.status(200).json({ reply });
      return;

    } catch (proError: any) {
      // 2. Pro එකේ Rate limit පැන්නොත් හෝ වෙනත් Error එකක් ආවොත් මෙතනට එනවා
      console.warn("⚠️ Gemini 1.5 Pro Failed or Rate Limited! Switching to Gemini 1.5 Flash...");
      console.error("Pro Error Details:", proError?.message || proError);

      // 3. දැන් autoම වේගවත් 'gemini-1.5-flash' එකෙන් වැඩේ කරලා දෙනවා
      const reply = await executeGeminiAgent("gemini-1.5-flash", message);
      
      res.status(200).json({ 
        reply,
        note: "Processed using fallback model due to high demand." // අවශ්‍ය නම් විතරක් Frontend එකට දැනුම් දෙන්න ගන්න පුළුවන්
      });
      return;
    }

  } catch (error: any) {
    // 4. Models දෙකම fail වුනොත් (උදා: Internet/API Key අවුල් නම්)
    console.error("❌ Both Gemini Models Failed:", error);
    res.status(500).json({ 
      error: "Internal Server Error", 
      details: error?.message || error 
    });
  }
};
